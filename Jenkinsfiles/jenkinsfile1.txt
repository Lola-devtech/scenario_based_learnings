pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                gitCheckout(
                branch: "main",
                url: "https://github.com/Lola-devtech/scenario_based_learnings.git"
                )
            }
        }
        
        stage('Deploy with Rollback') {
            steps {
                sh './deploy.sh --rollback-on-failure'
            }
        }
    }
}
